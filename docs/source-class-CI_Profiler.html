103:      *
104:      * This function cycles through the entire array of mark points and
105:      * matches any two points that are named identically (ending in "_start"

108:      *
109:      * @return  array

111:     protected function _compile_benchmarks()
112:     {
113:         $profile = array();
114:         $output = array();
115:         
116:         foreach ($this->CI->benchmark->marker as $key => $val)
117:         {
118:             // We match the "end" marker so that the list ends
119:             // up in the order that it was defined
120:             if (preg_match("/(.+?)_end/i", $key, $match))
121:             {
122:                 if (isset($this->CI->benchmark->marker[$match[1].'_end']) AND isset($this->CI->benchmark->marker[$match[1].'_start']))
123:                 {
124:                     $profile[$match[1]] = $this->CI->benchmark->elapsed_time($match[1].'_start', $key);
125:                 }
126:             }
127:         }
128: 
129:         // Build a table containing the profile data.

132:         foreach ($profile as $key => $val)
133:         {

453:     public static function get_file_size($size, $retstring = null) {
454:         // adapted from code at http://aidanlister.com/repos/v/function.size_readable.php
455:         $sizes = array('bytes', 'kB', 'MB', 'GB', 'TB', 'PB', 'EB', 'ZB', 'YB');
456: 
457:         if ($retstring === null) { $retstring = '%01.2f %s'; }
458: 
459:         $lastsizestring = end($sizes);
460: 
461:         foreach ($sizes as $sizestring) {
462:             if ($size < 1024) { break; }
463:                if ($sizestring != $lastsizestring) { $size /= 1024; }
464:         }
465:         
466:         if ($sizestring == $sizes[0]) { $retstring = '%01d %s'; } // Bytes aren't normally fractional
467:         return sprintf($retstring, $size, $sizestring);
468:     }
