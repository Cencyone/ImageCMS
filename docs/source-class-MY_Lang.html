 74:     public function load_gettext( $userlang = '', $codeset = '', $textdomain = 'lang', $path = '' ) {
 75:         if ( !isset($this->ci) )
 76:             $this->ci =& get_instance();

142:         if ( !isset($this->gettext_domain) )

146:             $sTranslate = ngettext($original, $aParams['plural'], $aParams['count']);

168:         for ( $i=1, $iMax = func_num_args(); $i<$iMax; $i++) {
169:             $arg = func_get_arg($i);
170:             if (is_array($arg)) {
171:                 foreach ($arg as $key => $sValue) {
172:                     $aTrad['%'.$key] = $sValue;
173:                 }
174:             } else {
175:                 $aTrad['%'.$key] = $arg;
